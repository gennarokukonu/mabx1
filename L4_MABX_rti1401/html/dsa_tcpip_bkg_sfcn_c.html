66static void mdlInitializeSizes(SimStruct *S)



67{



68    /* See sfuntmpl.doc for more details on the macros below */



69    



70    ssSetNumSFcnParams(S, NUM_PARAM);  /* Number of expected parameters */



71    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {



72        /* Return if number of expected != number of actual parameters */



73        return;



74    }



75    



76    ssSetNumContStates(S, 0);



77    ssSetNumDiscStates(S, 0);



78    



79    /* input ports */



80    if (!ssSetNumInputPorts(S, NUM_INPUTS)) return;



81    



82    /* input: INPUT_VALUES */



83    // ssSetInputPortDataType(S, INPUT_VALUES, SS_DOUBLE);



84    // ssSetInputPortWidth(S, INPUT_VALUES, mxGetPr(ssGetSFcnParam(S, IDX_PARAM_WIDTH))[0]);



85    // ssSetInputPortDirectFeedThrough(S, INPUT_VALUES, 1);



86    



87    /* output ports */



88    if (!ssSetNumOutputPorts(S, NUM_OUTPUTS)) return;



89    



90    ssSetNumSampleTimes(S, 1);



91    ssSetNumRWork(S, 0);



92    ssSetNumIWork(S, NUM_IWORK);



93    ssSetNumPWork(S, NUM_PWORK);



94    ssSetNumModes(S, 0);



95    ssSetNumNonsampledZCs(S, 0);



96    



97    ssSetOptions(S, 0);



98}

158static void mdlOutputs(SimStruct *S, int_T tid) {



159#ifndef MATLAB_MEX_FILE



160    void    **PWork  = ssGetPWork(S);



161    int_T    *IWork  = ssGetIWork(S);



162    



163    UInt32 irqStatus;



164    



165    /* if this is the first time: initialize lwip timers */



166    if (IWork[IDX_IWORK_FIRST]) {



167        IWork[IDX_IWORK_FIRST] = 0;



168        sys_timeouts_init();



169    }



170    



171    



172    



173    /* call background function of the netif driver */



174    // msg_info_printf(0,0,"Enter dsethif_background");



175    dsethif_background();



176    // msg_info_printf(0,0,"Enter sys_check_timeouts");



177    



178    /* run all time triggered actions of the TCP/IP stack. Must not be interrupted by other calls to the TCP/IP stack */



179    RTLIB_INT_SAVE_AND_DISABLE(irqStatus);



180    sys_check_timeouts();



181    RTLIB_INT_RESTORE(irqStatus);



182    //msg_info_printf(0,0,"Exit sys_check_timeouts");



183    



184    



185#endif /* MATLAB_MEX_FILE */



186}
