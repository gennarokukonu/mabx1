74static void mdlInitializeSizes(SimStruct *S)



75{



76    /* See sfuntmpl.doc for more details on the macros below */



77    



78    ssSetNumSFcnParams(S, NUM_PARAM);  /* Number of expected parameters */



79    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {



80        /* Return if number of expected != number of actual parameters */



81        return;



82    }



83    



84    ssSetNumContStates(S, 0);



85    ssSetNumDiscStates(S, 0);



86    



87    /* input ports */



88    if (!ssSetNumInputPorts(S, NUM_INPUTS)) return;



89    



90    /* input: INPUT_VALUES */



91    // ssSetInputPortDataType(S, INPUT_VALUES, SS_DOUBLE);



92    // ssSetInputPortWidth(S, INPUT_VALUES, mxGetPr(ssGetSFcnParam(S, IDX_PARAM_WIDTH))[0]);



93    // ssSetInputPortDirectFeedThrough(S, INPUT_VALUES, 1);



94    



95    /* output ports */



96    if (!ssSetNumOutputPorts(S, NUM_OUTPUTS)) return;



97    



98    /* output: IDX_OUTPUT_STATUS */



99    ssSetOutputPortDataType(S, IDX_OUTPUT_LINK_STATUS, SS_DOUBLE);



100    ssSetOutputPortWidth(S, IDX_OUTPUT_LINK_STATUS, 1);



101    



102    ssSetNumSampleTimes(S, 1);



103    ssSetNumRWork(S, 0);



104    ssSetNumIWork(S, NUM_IWORK);



105    ssSetNumPWork(S, NUM_PWORK);



106    ssSetNumModes(S, 0);



107    ssSetNumNonsampledZCs(S, 0);



108    



109    ssSetOptions(S, 0);



110}

113/* Function: mdlInitializeSampleTimes ======================================



114 * Abstract:



115 *    This function is used to specify the sample time(s) for your



116 *    S-function. You must register the same number of sample times as



117 *    specified in ssSetNumSampleTimes.



118 */



119static void mdlInitializeSampleTimes(SimStruct *S)



120{



121    ssSetSampleTime(S, 0, INHERITED_SAMPLE_TIME);



122    ssSetOffsetTime(S, 0, FIXED_IN_MINOR_STEP_OFFSET);



123



124#ifndef MATLAB_MEX_FILE



125    {



126        /* Network interface variables */



127        void          **PWork      = ssGetPWork(S);



128        struct ip_addr  ipaddr, netmask, gw;



129        struct netif   *netif;



130        struct dsethif *state;



131        double          startTime;



132        



133        /* used ethernet hardware */



134        int boardType = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_BOARDTYPE))[0];



135        int boardNo   = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_BOARDNO))[0];



136        int channelNo = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_CHANNELNO))[0];



137        



138        /* local ethernet configuration */



139        UInt32 uiIp = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_IPADDR))[0];



140        UInt32 uiNm = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_NETMASK))[0];



141        UInt32 uiGw = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_GATEWAY))[0];



142        



143        netif = malloc(sizeof(struct netif));



144        



145        /* Set network address variables */



146        IP4_ADDR(&ipaddr,   (uiIp >> 24) & 0xFF, (uiIp >> 16) & 0xFF, (uiIp >> 8) & 0xFF, uiIp & 0xFF);



147        IP4_ADDR(&netmask,  (uiNm >> 24) & 0xFF, (uiNm >> 16) & 0xFF, (uiNm >> 8) & 0xFF, uiNm & 0xFF);



148        IP4_ADDR(&gw,       (uiGw >> 24) & 0xFF, (uiGw >> 16) & 0xFF, (uiGw >> 8) & 0xFF, uiGw & 0xFF);



149        



150#if NO_SYS



151        /* The lwIP single-threaded core: initialize the network stack */



152        lwip_init();



153#else



154#error "!NO_SYS initialization not implemented"



155        /* lwIP in a multi-threaded system: initialize the network stack */



156        tcpip_init(tcpip_init_done, tcpip_init_done_param);



157        /* implicitly calls lwip_init();



158           start new thread



159           calls tcpip_init_done(tcpip_init_done_param); when tcpip init done



160           tcpip_init_done and tcpip_init_done_param are user-defined (may be NULL). */



161        /* todo: wait for tcpip_init_done() ? */



162#endif



163        



164        /* Bring up the network interface */



165        /* Hint: netif_init(); was already called by lwip_init(); above */



166        state = dsethif_create(boardType, boardNo, channelNo);



167        netif_add(netif, &ipaddr, &netmask, &gw, state, dsethif_init, ethernet_input);



168        



169        /* bring up interface */



170        netif_set_default(netif);



171        netif_set_up(netif);



172        



173        /* save pointer to netif */



174        PWork[IDX_PWORK_NETIF] = netif;



175    }



176    



177#endif /* MATLAB_MEX_FILE */



178}

222static void mdlOutputs(SimStruct *S, int_T tid) {



223#ifndef MATLAB_MEX_FILE



224    void      **PWork            = ssGetPWork(S);



225    int_T      *IWork            = ssGetIWork(S);



226    Float64    *outputLinkStatus = ssGetOutputPortRealSignal(S, IDX_OUTPUT_LINK_STATUS);



227    



228    struct netif   *netif = PWork[IDX_PWORK_NETIF];



229    



230    /* check if link is established */



231    if (netif_is_link_up(netif)) {



232        *outputLinkStatus = 1;



233    } else {



234        *outputLinkStatus = 0;



235    }



236    



237    /* call irq function of the netif driver because netif_set_up tries to send a packet



238     * but in mdlStart interrupts are disabled. So this interrupt hangs and no further IRQ



239     * can be transmitted -> deadlock 



240     * Only necessary if device is initialized while mdlStart (see low_level_init of driver) */



241#if 0



242    if (IWork[IDX_IWORK_FIRST]) {



243        UInt32 status;



244        IWork[IDX_IWORK_FIRST] = 0;



245        



246        /* don't know if irq function is reentrant. */



247        RTLIB_INT_SAVE_AND_DISABLE(status);



248        dsethif_irq();



249        RTLIB_INT_RESTORE(status);



250    }



251#endif



252    



253#endif /* MATLAB_MEX_FILE */
