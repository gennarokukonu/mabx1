93static void mdlInitializeSizes(SimStruct *S)



94{

96    boolean_T legacyMode;

99    ssSetNumSFcnParams(S, NUM_PARAM);  /* Number of expected parameters */



100    if (ssGetNumSFcnParams(S) != ssGetSFcnParamsCount(S)) {



101        /* Return if number of expected != number of actual parameters */



102        return;



103    }



104    



105    ssSetNumContStates(S, 0);



106    ssSetNumDiscStates(S, 0);

108    



109    /* check parameters */



110    value = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_LOCAL_PORT))[0];



111    if (value < 0 || value > 65535) {



112        ssSetErrorStatus(S, "Parameter local port out of range!");

246    /* pbuf queue */



247    volatile UInt32 pbufsInQueue;



248    volatile UInt32 pbufQueueWrite;



249    volatile UInt32 pbufQueueRead;



250    



251    /* udp socket info */



252    UInt16 localPort;



253    ip_addr_t remoteIpAddr;



254    UInt16 remotePort;



255    



256    /* udp status */



257    volatile UInt32 udpConfigured;      /* we allow only one connection to a listen socket */

259    



260    struct pbuf *pbufs[TCPIP_MAX_RCV_PBUFS];

264static void dsa_udpip_recv(void *arg, struct udp_pcb *upcb, struct pbuf *p, struct ip_addr *addr, u16_t port) {



265    



266    UInt32 status;



267    



268    /* get pointer to the additional structure */



269    struct dsaUdpSocketArg* udpSocketArg = arg;



270    



271    /* check if buffer can be put in the queue */



272    if (udpSocketArg->pbufsInQueue >= TCPIP_MAX_RCV_PBUFS) {



273        msg_info_printf(0,0, "UDP: No more free space in the block's list of buffer objects. Message lost.");



274        return;



275    }



276    



277    /* disable interrupts. Funktion is not reentrant and reading at the same time is not safe */



278    RTLIB_INT_SAVE_AND_DISABLE(status);



279    



280    /* put the pbuf in the queue and increment write pointer*/



281    udpSocketArg->pbufs[udpSocketArg->pbufQueueWrite] = p;



282    



283    /* goto next pbuf in chain and wrap around if necessary */



284    udpSocketArg->pbufQueueWrite++;



285    if (udpSocketArg->pbufQueueWrite >= TCPIP_MAX_RCV_PBUFS) {



286        udpSocketArg->pbufQueueWrite = 0;



287    }



288    



289    /* update sizes */



290    udpSocketArg->pbufsInQueue++;



291    



292    RTLIB_INT_RESTORE(status);



293    



294    return;



295}

303  /* Function: mdlStart ====================================================



304   * Abstract:



305   *    This function is called once at start of model execution. If you



306   *    have states that should be initialized once, this is the place



307   *    to do it.



308   */



309static void mdlStart(SimStruct *S) {



310#ifndef MATLAB_MEX_FILE



311    void    **PWork           = ssGetPWork(S);



312    int_T    *IWork           = ssGetIWork(S);



313    UInt32    ipAddr;



314    err_t     result;



315    



316    struct dsaUdpSocketArg *udpSocketArg;



317    



318    /* get memory for local socket structure */



319    udpSocketArg = malloc(sizeof(struct dsaUdpSocketArg));



320    if (udpSocketArg == NULL) {



321        msg_error_printf(0,0, "TCP/IP (UDP): Out of memory");



322    }



323    memset(udpSocketArg, 0, sizeof(struct dsaUdpSocketArg));



324    PWork[PWORK_UDP_SOCKET_ARG] = udpSocketArg;



325    



326    /* get param local port */



327    udpSocketArg->localPort = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_LOCAL_PORT))[0];



328    



329    /* open new udp socket */



330    udpSocketArg->udpSocPcb = udp_new();



331    if (udpSocketArg->udpSocPcb == NULL) {



332        msg_error_printf(0,0, "Could not create new socket object for UDP/IP - No Setup block present?");



333        RTLIB_EXIT(1);



334    }



335    



336    /* bind to local address. We have only one local address so we use IP_ADDR_ANY */



337    result = udp_bind(udpSocketArg->udpSocPcb, IP_ADDR_ANY, udpSocketArg->localPort);



338    if (result != ERR_OK) {



339        msg_error_printf(0,0, "UDP/IP Could not bind to port: %d", udpSocketArg->localPort);



340        RTLIB_EXIT(1);



341    }



342    



343    /* configure call function for receiving packets */



344    udp_recv(udpSocketArg->udpSocPcb, dsa_udpip_recv, udpSocketArg);



345    



346    /* set remote ip address and port for sending */



347    ipAddr = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_REMOTE_ADDR))[0];



348    IP4_ADDR(&udpSocketArg->remoteIpAddr, (ipAddr >> 24) & 0xFF, (ipAddr >> 16) & 0xFF, (ipAddr >> 8) & 0xFF, ipAddr & 0xFF);



349    udpSocketArg->remotePort = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_REMOTE_PORT))[0];



350    



351    /* are we have successfully configured the upd socket */



352    udpSocketArg->udpConfigured = 1;



353    



354    



355#endif /* MATLAB MEX FILE */



356}



357#endif /*  MDL_START */

360static void mdlOutputs(SimStruct *S, int_T tid) {



361#ifndef MATLAB_MEX_FILE



362    void      **PWork             = ssGetPWork(S);



363    //Int32       inputEnableRead   = (Int32)  *(ssGetInputPortRealSignalPtrs(S, IDX_INPUT_ENABLE_RECV)[0]);



364



365    Bool        inputEnableRead   = **((Bool   **)ssGetInputPortSignalPtrs(S, IDX_INPUT_ENABLE_RECV));



366    UInt32      inputBytesToWrite = **((UInt32 **)ssGetInputPortSignalPtrs(S, IDX_INPUT_BYTES_SEND));



367



368



369    UInt32    **inputStream       = 0;



370    UInt8     **inputStreamByte   = 0;



371    Float64    *outputStatus      = 0;



372    UInt32     *outputStatusUint  = 0;



373    Float64    *outputRxSize      = 0;



374    UInt32     *outputRxSizeUint  = 0;



375    UInt32     *outputStream      = 0;



376    UInt8      *outputStreamByte  = 0;



377    UInt32      irqStatus;



378    err_t       result;



379    int         i;



380    int         numPbufsInQueue;



381    Bool        legacyMode        = (Bool) mxGetPr(ssGetSFcnParam(S, IDX_PARAM_LEGACY_MODE))[0];



382



383    /* get pointer to the additional structure */



384    struct dsaUdpSocketArg* udpSocketArg = PWork[PWORK_UDP_SOCKET_ARG];



385    



386    if (legacyMode) {



387    	inputStream  = (UInt32 **)ssGetInputPortSignalPtrs(S, IDX_INPUT_32STREAM);



388        outputStream = (UInt32 *) ssGetOutputPortSignal(S, IDX_OUTPUT_32STREAM);



389        outputStatus = (Float64 *)ssGetOutputPortRealSignal(S, IDX_OUTPUT_STATUS);



390        outputRxSize = (Float64 *)ssGetOutputPortRealSignal(S, IDX_OUTPUT_SIZE);



391    }



392    else {



393    	inputStreamByte  = (UInt8 **)ssGetInputPortSignalPtrs(S, IDX_INPUT_32STREAM);



394    	outputStreamByte = (UInt8 *) ssGetOutputPortSignal(S, IDX_OUTPUT_32STREAM);



395    	outputStatusUint = (UInt32 *) ssGetOutputPortSignal(S, IDX_OUTPUT_STATUS);



396    	outputRxSizeUint = (UInt32 *) ssGetOutputPortSignal(S, IDX_OUTPUT_SIZE);



397    }



398



399    /**************\



400    * Transmit data*



401    \**************/



402    if (udpSocketArg->udpConfigured && inputBytesToWrite > 0) {



403        int             sendSize, bytesCopied, bytesToCopy;



404        struct pbuf    *pData, *p;



405        UInt32         *pData_32;



406        UInt8          *pData_8;



407        UInt32          inputIndex;



408        err_t           result;



409        



410        /* limit size to the maximum input size */



411        sendSize = inputBytesToWrite;



412        if (sendSize > mxGetPr(ssGetSFcnParam(S, IDX_PARAM_INPUT_SIZE))[0]) {



413            sendSize = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_INPUT_SIZE))[0];



414        }



415        



416        /*



417         * try to get a buffer to copy the packet (All header up to IP + UDP Header)



418         * We may get a pbuf chain but it is guaranteed the length of all pbufs execpt



419         * the last is alligned to 4 bytes. This makes data copy with byte order correction



420         * a lot easier.



421         */



422        pData = pbuf_alloc(PBUF_IP, UDP_HLEN + sendSize, PBUF_POOL);



423        



424        /* check if data can be written */



425        if (pData != NULL) {



426            



427            /* add room for UDP header. udp_sendto will try to get room for header. Otherwise it must allocate an new pbuf for the header */



428            result = pbuf_header(pData, -UDP_HLEN);



429            



430            bytesCopied = 0;



431            inputIndex = 0;



432            p = pData;



433            



434            while (bytesCopied < sendSize) {



435                /* calculate bytes to copy into this pbuf */



436                bytesToCopy = sendSize - bytesCopied;



437                if (bytesToCopy > p->len) {



438                    bytesToCopy = p->len;



439                }



440                



441                if (legacyMode){



442					/* copy full 32bit words to output port */



443					pData_32 = (UInt32 *)p->payload;



444					for (i = 0; i < (bytesToCopy >> 2); i++) {



445						pData_32[i] = SWAPL_MOTOROLA(*inputStream[inputIndex]);



446						inputIndex++;



447					}



448					bytesCopied += (bytesToCopy >> 2) * 4;



449



450					/* copy up to 3 additional bytes */



451					pData_8 = (UInt8*)&pData_32[i];



452					switch(bytesToCopy % 4) {



453						case 3:



454							pData_8[2] = (*inputStream[inputIndex] >> 16) & 0xFF;



455							/* fall through */



456						case 2:



457							pData_8[1] = (*inputStream[inputIndex] >>  8) & 0xFF;



458							/* fall through */



459						case 1:



460							pData_8[0] = *inputStream[inputIndex] & 0xFF;



461							bytesCopied += bytesToCopy % 4;



462



463							/* check if these are the last bytes */



464							if (bytesCopied != sendSize) {



465								msg_warning_printf(0,0, "UDP: Internal Error: Copied <4 bytes not at the buffer end. Copy data to Packet is stopped after %d bytes.", bytesCopied);



466								sendSize = bytesCopied;



467							}



468					}



469                } else {



470					/* copy bytes */



471                	pData_8 = (UInt8 *) p->payload;



472					for (i = 0; i < bytesToCopy; i++) {



473						pData_8[i] = *inputStreamByte[inputIndex];



474						inputIndex++;



475					}



476					bytesCopied += bytesToCopy;



477                }



478                



479                /* goto next pbuf in chain */



480                p = p->next;



481            }



482            



483            /* try to send data */



484            RTLIB_INT_SAVE_AND_DISABLE(irqStatus);



485            



486            //msg_info_printf(0,0,"Enter udp_sendto");



487            result = udp_sendto(udpSocketArg->udpSocPcb, pData, &udpSocketArg->remoteIpAddr, udpSocketArg->remotePort);



488            // msg_info_printf(0,0,"Exit udp_sendto");



489            



490            RTLIB_INT_RESTORE(irqStatus);



491            



492            if(legacyMode){



493				if (result != ERR_OK) {



494					outputStatus[OUT_STATUS_TXERROR] = result;



495				} else {



496					outputStatus[OUT_STATUS_TXERROR] = 0;



497				}



498            } else {



499            	if (result != ERR_OK) {



500					outputStatusUint[OUT_STATUS_TXERROR] = result;



501				} else {



502					outputStatusUint[OUT_STATUS_TXERROR] = 0;



503				}



504            }



505            



506            /* free pbuf object */



507            pbuf_free(pData);



508        }



509    }



510    



511    



512    /**************\



513    * Receive Data  *



514    \**************/



515    



516    /* read number of message in queue only once so it does not change if a packet arrives during execution.



517     * Otherwise you might get a "No packet received" status and an amount of one packet is in the buffer,



518     * if a new packet arrives shortly after checking for msgs in the buffer */



519    numPbufsInQueue = udpSocketArg->pbufsInQueue;



520    



521    if (inputEnableRead) {



522        /* check for a message in the queue */



523        if (numPbufsInQueue == 0) {



524        	if (legacyMode){



525        		outputStatus[OUT_STATUS_RXNEWDATA] = 0;



526        	}



527        	else {



528        		outputStatusUint[OUT_STATUS_RXNEWDATA] = 0;



529        	}



530        /* there is at least one new message available */



531        } else {



532            int             recvSize, bytesCopied, bytesToCopy;



533            struct pbuf    *pData, *p;



534            UInt32         *pData_32;



535            UInt8          *pData_8;



536            UInt32         outputIndex;



537            



538            /* get pbuf object from chain */



539            RTLIB_INT_SAVE_AND_DISABLE(irqStatus);



540            



541            pData = udpSocketArg->pbufs[udpSocketArg->pbufQueueRead];



542            udpSocketArg->pbufQueueRead++;



543            if (udpSocketArg->pbufQueueRead >= TCPIP_MAX_RCV_PBUFS) {



544                udpSocketArg->pbufQueueRead = 0;



545            }



546            udpSocketArg->pbufsInQueue--;



547            numPbufsInQueue--;



548            



549            RTLIB_INT_RESTORE(irqStatus);



550            



551            /* get length from first pbuf if this is a chain */



552            recvSize = pData->tot_len;



553



554            if (legacyMode) {



555            	outputStatus[OUT_STATUS_RXNEWDATA] = 1;



556				*outputRxSize = recvSize;



557            } else {



558            	outputStatusUint[OUT_STATUS_RXNEWDATA] = 1;



559            	*outputRxSizeUint = recvSize;



560            }



561



562            /* limit length to maximum receive length */



563            if (recvSize > mxGetPr(ssGetSFcnParam(S, IDX_PARAM_OUTPUT_SIZE))[0]) {



564                recvSize = mxGetPr(ssGetSFcnParam(S, IDX_PARAM_OUTPUT_SIZE))[0];



565            }



566            



567            outputIndex = 0;



568            bytesCopied = 0;



569            p = pData;



570            



571            /* process all pbufs in the chain */



572            while (bytesCopied < recvSize) {



573                



574                /* calculate bytes to copy into this pbuf */



575                bytesToCopy = recvSize - bytesCopied;



576                if (bytesToCopy > p->len) {



577                    bytesToCopy = p->len;



578                }



579                



580                if (legacyMode) {



581					/* copy 32 bit data from this element */



582					pData_32 = (UInt32 *)p->payload;



583					for (i = 0; i < (bytesToCopy >> 2); i++) {



584						outputStream[outputIndex] = SWAPL_MOTOROLA(pData_32[i]);



585						outputIndex++;



586					}



587					bytesCopied += (bytesToCopy >> 2) * 4;



588



589					pData_8 = (UInt8*)&pData_32[i];



590                    



591					if (bytesToCopy % 4 != 0) {



592                        outputStream[outputIndex] = 0;



593                    }



594                    



595					switch(bytesToCopy % 4) {



596						case 3:



597							outputStream[outputIndex] |= pData_8[2] << 16;



598							bytesCopied++;



599							/* fall through */



600						case 2:



601							outputStream[outputIndex] |= pData_8[1] <<  8;



602							bytesCopied++;



603							/* fall through */



604						case 1:



605							outputStream[outputIndex] |= pData_8[0];



606							bytesCopied++;



607



608							if (bytesCopied != recvSize) {



609								msg_warning_printf(0,0, "UDP: Internal Error: Copied <4 bytes not at the buffer end. Copy data to output is stopped after %d bytes.", bytesCopied);



610								bytesCopied = recvSize;



611							}



612					}



613                } else {



614                	pData_8 = (UInt8 *) p->payload;



615					for (i = 0; i < bytesToCopy; i++) {



616						outputStreamByte[outputIndex] = pData_8[i];



617						outputIndex++;



618					}



619					bytesCopied += bytesToCopy;



620                }



621                



622                /* goto next element */



623                p = p->next;



624            }



625            



626            /* free pbuf object */



627            pbuf_free(pData);



628        }



629        



630    } else { /* if (inputEnableRead) */



631    	if (legacyMode){



632			outputStatus[OUT_STATUS_RXNEWDATA] = 0;



633    	} else {



634    		outputStatusUint[OUT_STATUS_RXNEWDATA] = 0;



635    	}



636    }

639    if (legacyMode) {



640		outputStatus[OUT_STATUS_REMAININGMSGS] = numPbufsInQueue;



641    } else {



642    	outputStatusUint[OUT_STATUS_REMAININGMSGS] = numPbufsInQueue;
